days = [1,2,3,4,5]
enfield = [50,40,70,80,20]
honda = [80,20,20,50,60]
yamaha = [70,20,60,40,60]
ktm = [80,20,20,50,60]
activities = [enfield, honda, yamaha, ktm]
bike = ['Enfield', 'Honda', 'Yamaha', 'KTM']
colors = ['r', 'y', 'g', 'b', 'm']
explode = (0,0,0,0,0)
legends = 'list of bike details'
title = 'pie plot'

plt.style.use('fivethirtyeight')
#create a figure
fig=plt.figure()
#fig.title("Pie Plot")
fig.legend("list of bike details")
#define subplots and their positions in figure
plt1=fig.add_subplot(221)
plt2=fig.add_subplot(222)
plt3=fig.add_subplot(223)
plt4=fig.add_subplot(224)
#plotting points on each subplot
plt1.pie(days,labels=activities[0],colors=colors,startangle=90,shadow=True,explode=explode,radius=1.2,autopct='%1.1f%%')
plt1.set_title(bike[0])

plt2.pie(days,labels=activities[1],colors=colors,startangle=90,shadow=True,explode=explode,radius=1.2,autopct='%1.1f%%')
plt2.set_title(bike[1])

plt3.pie(days,labels=activities[2],colors=colors,startangle=90,shadow=True,explode=explode,radius=1.2,autopct='%1.1f%%')
plt3.set_title(bike[2])

plt4.pie(days,labels=activities[3],colors=colors,startangle=90,shadow=True,explode=explode,radius=1.2,autopct='%1.1f%%')
plt4.set_title(bike[3])
fig.subplots_adjust(hspace=.5,wspace=0.5)

plt.show()
